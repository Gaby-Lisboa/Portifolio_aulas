Exercícios de JavaScript

Exercício 1: Tabuada de um número

// Código
var numero = Number(document.getElementById("numeroTabuada").value);
var resultado = "Tabuada do " + numero + ":\n";
for (var i = 1; i <= 10; i++) {
resultado += `${numero} x ${i} = ${numero * i}\n`;
}
document.getElementById("result1").innerText = resultado;

Exercício 2: Jogo de Pedra, Papel e Tesoura

// Código
var opcoes = ["pedra", "papel", "tesoura"];
var escolhaComputador = opcoes[Math.floor(Math.random() * 3)];
var escolhaUsuario = "pedra"; // fixo para este exemplo
var resultado = `Você escolheu: ${escolhaUsuario}\n`;
resultado += `Computador escolheu: ${escolhaComputador}\n`;
if (escolhaComputador === "pedra") resultado += "Empate!";
else if (escolhaComputador === "papel") resultado += "Você perdeu!";
else resultado += "Você ganhou!";
document.getElementById("result2").innerText = resultado;

Exercício 3: Calcular Fatorial

// Código
var numeroFatorial = Number(document.getElementById("numeroFatorial").value);
var fatorial = 1;
for (var i = 1; i <= numeroFatorial; i++) {
fatorial *= i;
}
document.getElementById("result3").innerText = `O fatorial de ${numeroFatorial} é ${fatorial}`;
