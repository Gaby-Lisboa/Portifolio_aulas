Exercício 1: Verificar Notas

// Código
var nota1 = Number(document.getElementById("nota1").value);
var nota2 = Number(document.getElementById("nota2").value);
var media = (nota1 + nota2) / 2;
var resultado1 = `A média das notas é ${media}`;
if (media >= 6) resultado1 += " (acima da média)";
if (nota1 % 2 === 0) resultado1 += `; A nota 1 (${nota1}) é par`;
if (nota2 % 2 === 0) resultado1 += `; A nota 2 (${nota2}) é par`;

Exercício 2: Calcular o IMC

// Código
var peso = Number(document.getElementById("peso").value);
var altura = Number(document.getElementById("altura").value);
var imc = peso / (altura * altura);
document.getElementById("result2").innerText = `O IMC é ${imc.toFixed(2)}`;

Exercício 3: Verificar Ano Bissexto

// Código
var ano = Number(document.getElementById("ano").value);
var bissexto = (ano % 4 === 0 && ano % 100 !== 0) || (ano % 400 === 0);
if (bissexto) document.getElementById("result3").innerText = `${ano} é bissexto`;
else document.getElementById("result3").innerText = `${ano} não é bissexto`;
